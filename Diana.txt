data = readtable('C:\Users\adith\OneDrive\Desktop\adithya\jm1.csv');
features = data{:, 1:end-1};
distance = 'euclidean'; 
linkage = 'single'; 
threshold = 0.2; 
clusters = clusterdata(features, 'criterion', 'distance', 'distance', distance, 'linkage', linkage, 'cutoff', threshold);
figure;
gscatter(features(:,1), features(:,2), clusters);
title('DIANA Clustering');
xlabel('Feature 1');
ylabel('Feature 2');
legend('Cluster 1', 'Cluster 2', 'Cluster 3', 'Cluster 4');
